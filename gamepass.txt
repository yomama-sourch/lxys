-- Tab Switching
local function updateTabs(activeTab)
    if activeTab == ListenerTab then
        ListenerTab.TextColor3 = Color3.fromRGB(255, 255, 255)
        ActionTab.TextColor3 = Color3.fromRGB(100, 100, 100)
        listenerTabFrame.Visible = true
        actionTabFrame.Visible = false
    elseif activeTab == ActionTab then
        ActionTab.TextColor3 = Color3.fromRGB(255, 255, 255)
        ListenerTab.TextColor3 = Color3.fromRGB(100, 100, 100)
        actionTabFrame.Visible = true
        listenerTabFrame.Visible = false
    end
end

ListenerTab.MouseButton1Click:Connect(function() updateTabs(ListenerTab) end)
ActionTab.MouseButton1Click:Connect(function() updateTabs(ActionTab) end)

-- Original Drag Logic (Dragify)
local UIS = game:GetService("UserInputService")
function dragify(Frame)
    dragToggle = nil
    dragSpeed = 0.15
    dragInput = nil
    dragStart = nil
    dragPos = nil
    function updateInput(input)
        Delta = input.Position - dragStart
        Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + Delta.X, startPos.Y.Scale, startPos.Y.Offset + Delta.Y)
        game:GetService("TweenService"):Create(Frame, TweenInfo.new(0.15), {Position = Position}):Play()
    end
    Frame.InputBegan:Connect(function(input)
        if (input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch) and UIS:GetFocusedTextBox() == nil then
            dragToggle = true
            dragStart = input.Position
            startPos = Frame.Position
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    dragToggle = false
                end
            end)
        end
    end)
    Frame.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
            dragInput = input
        end
    end)
    game:GetService("UserInputService").InputChanged:Connect(function(input)
        if input == dragInput and dragToggle then
            updateInput(input)
        end
    end)
end
dragify(mainbg)
